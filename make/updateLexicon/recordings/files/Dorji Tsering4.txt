ཏིག་ཙི་ གྲ་རྒྱས་པོ་ བྱེད་དགོས་ རེད་ ད།  མདུན་ཅོག་ དང་ དེ འི་ སྒང་ ལ་ ཁ་ལག་ ཟ་ དུས་ དགོས་ ཀྱི་ མ་རེད་བ ས།  སྐེ་ ལ་ འདོགས་ཡ་ དེ་འདྲ འི་ དགོས་ ཀྱི་ མ་རེད་བ།  ཇ་ཁང་ ཆེན་པོ་ དེ་འདྲ འི་ ནང་དུ་ དགོས་ ཀྱི་ ཡོད་པ།  ང་ཚོ་ དེ་ ད་ག་སེ་ རེད་ མི་འདུག་ ག  
ཅོག་ཙེ་ སྒང་ ལ་ བཀབ་ཡ་ ལབ་ ཡིན་པ ས།   
འོ་ དེ།  དེ་ རིན་པོ་ཆེ འི་ གཉེར་བ ས་ གྲ་སྒྲིག་ བྱེད་ཡ་ ཡིན་གྱི་རེད།    
ད་ སྐར་མ་ ལྔ འི་ རྗེས་སུ་ ངེད་གཉིས་ འགྲོ་ དགོས་ རེད་བ།  སྐར་མ་ ལྔ་ ཟིན་ འདུག་ ད།  དངོས་གནས་ ཟིན་ ཡོད་ས་ རེད།  ད་ སྐར་མ་ བཅུ་ ཆགས་ ཚར་ འདུག  སྡིག་པ་ཁུར་ ངེད་གཉིས་ འགྲོ་ མ་གཏོགས།
